<script>
  let deferredPrompt;

  // Función para verificar si la app ya está instalada
  function isAppInstalled() {
    return window.navigator.standalone === true || window.matchMedia('(display-mode: standalone)').matches;
  }

  // Función para detectar dispositivos iOS (incluye iPadOS)
  function isIOS() {
    return (
      /iPhone|iPad|iPod/.test(navigator.userAgent) ||
      (navigator.platform === 'MacIntel' && navigator.maxTouchPoints > 1)
    );
  }

  // Función para detectar macOS (excluye iPadOS)
  function isMac() {
    return (
      navigator.platform.indexOf('Mac') >= 0 &&
      !(navigator.platform === 'MacIntel' && navigator.maxTouchPoints > 1)
    );
  }

  // Escuchar el evento beforeinstallprompt (para Android y navegadores compatibles)
  window.addEventListener('beforeinstallprompt', (e) => {
    e.preventDefault();
    deferredPrompt = e;
  });

  // Esperar a que el DOM esté completamente cargado
  document.addEventListener('DOMContentLoaded', () => {
    // Seleccionar el botón del slider por su atributo href único
    const installBtn = document.querySelector('a[href="#installPwa"]');
    if (installBtn) {
      installBtn.addEventListener('click', async (event) => {
        event.preventDefault();

        // Verificar si la app ya está instalada en cualquier plataforma
        if (isAppInstalled()) {
          alert('La aplicación ya está instalada.');
          return;
        }

        if (deferredPrompt) {
          // Mostrar el prompt nativo para instalar la app (Android, Chrome, Edge, etc.)
          deferredPrompt.prompt();
          const { outcome } = await deferredPrompt.userChoice;
          console.log('Respuesta del usuario:', outcome);
          deferredPrompt = null;
        } else {
          // Fallback para navegadores sin beforeinstallprompt
          if (isIOS()) {
            alert('Presioná el ícono de compartir y elegí "Añadir a Inicio".');
          } else if (isMac()) {
            alert('Presioná el ícono de compartir y "Añadir al Dock".');
          } else {
            alert('La instalación de esta aplicación no es compatible con este navegador.');
          }
        }
      });
    } else {
      console.error('No se encontró el botón de instalación. Asegúrate de que la URL del botón sea "#installPwa".');
    }
  });
</script>
